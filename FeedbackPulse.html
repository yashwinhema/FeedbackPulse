  } catch(e) { useMock = true; }

}

// ─── CORE TRACKING FUNCTION ────────────────────────────────────────

function mp(event, props = {}) {

  const enriched = {

    ...props,

    timestamp: new Date().toISOString(),

    app: 'FeedbackPulse',

    user_plan: currentUser?.plan || 'anonymous',

    session_id: sessionId,

    // Data health fields

    has_user_id: !!currentUser?.id,

    has_plan: !!currentUser?.plan,

    schema_version: '1.0'

  };


  // Fire to real Mixpanel if token set

  if (!useMock) {

    try { mixpanel.track(event, enriched); } catch(e) {}

  }


  // Always log locally

  logEvent(event, enriched);

  console.log(`[MP] ${event}`, enriched);

}


function mpIdentify(userId, traits = {}) {

  if (!useMock) {

    try {

      mixpanel.identify(userId);

      mixpanel.people.set({ ...traits, $last_login: new Date() });

    } catch(e) {}

  }

  logEvent('identify', { distinct_id: userId, ...traits }, 'identify');

}

// ─── EVENT LOG ────────────────────────────────────────────────────

function logEvent(name, props, type='track') {

  eventLog.unshift({ name, props, type, time: new Date().toLocaleTimeString() });

  if (eventLog.length > 50) eventLog.pop();

  renderEventLog();

}


function renderEventLog() {

  const el = document.getElementById('event-entries');

  if (!el) return;

  el.innerHTML = eventLog.map(e => `

    <div class="event-item">

      <div class="event-dot ${e.type}"></div>

      <div style="flex:1">

        <div class="event-name">${e.name}</div>

        <div class="event-props">${JSON.stringify(e.props, null, 0).slice(0,120)}...</div>

      </div>

      <div class="event-time">${e.time}</div>

    </div>`).join('') || '<div style="color:var(--muted);font-size:13px">No events yet. Interact with the app!</div>';

}


// ─── HELPERS ──────────────────────────────────────────────────────

function animateCount(id, target, prefix='') {

  const el = document.getElementById(id); if (!el) return;

  let cur = 0; const step = Math.ceil(target / 40);

  const t = setInterval(() => {

    cur = Math.min(cur + step, target);

    el.textContent = prefix + cur.toLocaleString();

    if (cur >= target) clearInterval(t);

  }, 30);

}

}
